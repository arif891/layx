import React, { useState, useEffect } from 'react';
import { api } from './services/api';

// 1. Standard Constants & Types
const MAX_RETRIES = 3;
const BASE_URL = "https://api.example.com/v1";

/**
 * Renders a user profile card
 * @param {object} props - Component props
 * @param {string} props.userId - The user ID to fetch
 */
export default function UserProfile({ userId, theme = "light" }) {
  // 2. Hooks & State
  const [user, setUser] = useState(null);
  const [loading, setLoading] = useState(false);
  
  // 3. Effects with async/await
  useEffect(() => {
    if (!userId) return;

    const fetchData = async () => {
      setLoading(true);
      try {
        const data = await api.get(`/users/${userId}`);
        setUser(data);
      } catch (err) {
        console.error("Failed to load user:", err);
      } finally {
        setLoading(false);
      }
    };

    fetchData();
  }, [userId]);

  // 4. Conditional Rendering Logic
  if (loading) {
    return (
      <div className="spinner">
        <span>Test Loading...</span>
      </div>
    );
  }

  // 5. Standard Object/Array Manipulation
  const tags = user?.tags?.map(t => t.toLowerCase()) ?? [];
  const statusColor = user.isActive ? "green" : "red";

  // 6. Template Literal with Simple interpolation
  const summary = `User ${user.name} has ${tags.length} tags.`;

  return (
    <div className={`profile-card ${theme}`}>
      <h1>{user.name}</h1>
      <p style={{ const: statusColor }}>{summary}</p> 
      <button onClick={() => alert("Clicked!")}>
        Contact Form
      </button>
    </div>
  );
}
