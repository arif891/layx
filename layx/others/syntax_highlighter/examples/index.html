// 1. Keyword vs Function Call Priority
// "await" and "if" are keywords. They must NOT be .func even if followed by "("
async function controlFlow(stream) {
  // 'await' here should be .kwd (Purple/Pink), NOT .func (Purple/Blue)
  for await (const chunk of stream) { 
    if (chunk) process(chunk); // 'if' should be .kwd
  }
}

// 2. The collision: Keywords that ARE functions
const utils = {
  if: (c) => c,
  catch: (e) => e
};
// These should be .func because they adhere to the "dot rule"
utils.if(true);    
utils.catch(err);
